CREATE TABLE IF NOT EXISTS public.lotacao
(
    lot_id integer NOT NULL DEFAULT nextval('lotacao_lot_id_seq'::regclass),
    pes_id integer,
    unid_id integer,
    lot_data_lotacao date,
    lot_data_remocao date,
    lot_portaria character varying(100) COLLATE pg_catalog."default",
    CONSTRAINT lotacao_pkey PRIMARY KEY (lot_id),
    CONSTRAINT lotacao_pes_id_fkey FOREIGN KEY (pes_id)
        REFERENCES public.pessoa (pes_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT lotacao_unid_id_fkey FOREIGN KEY (unid_id)
        REFERENCES public.unidade (unid_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
)

TABLESPACE pg_default;
